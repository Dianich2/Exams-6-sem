// тут даны небольшие пояснения к нашему конфигу для Squid
// (ну потому что с NGINX уже был опыт и не маленький, а вот Squid первый раз пробуем, нужно разобраться)


http_port 3128                          // слушает 3128 порт
acl all src all                         // Создать ACL с именем all. Тип проверки - src, то есть IP-адрес источника. Значение - all, то есть любые IP-адреса
http_access allow all                   // Разрешить HTTP-доступ всем, кто попадает под ACL all
access_log /var/log/squid/access.log    // Путь к журналу обращений
cache_log /var/log/squid/cache.log      // Путь к журналу самого Squid. Здесь будут написаны события работы самого Squid
visible_hostname squid-proxy            // Это имя, которым Squid представляется в служебных сообщениях
