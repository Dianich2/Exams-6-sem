// пишем приложение на go

// собираем его через WSL (так будет удобнее, чем писать всякие переменные окружения)
go build -o server-go

// создаем dockerfile для создания образа нашего приложения(прилагается со всем остальным добром)

// создаем тестовые и простенькие(а че парится по фронту) html, css, js(это типо иммитация, если так можно сказать, нашего bundle)

// дальше нужно поработать над HTTPS, а более конкретно нам нужны сертификат и ключ(ну тоже, чтобы было проще, делаем это все через WSL)
openssl req -x509 \
-newkey rsa:2048 \
-keyout key.pem \
-out cert.pem \
-days 365 \
-nodes

// создаем конфиг для nginx(по сути стандартный конфиг, но там нужно настроить всякое, чтобы мы могли показать и статику, и проксирование, и https; тоже прилагается)

// создаем файлик docker-compose.yml(упрощаем себе жизнь; тоже прилагается)

// запускаем наши контейнеры(или же без -d, чтобы не в фоне (это на следующий раз docker compose up))
docker compose up --build -d


// для проверки запуска html(там через кнопочку проверяем back)
https://localhost/

// для проверки, что отдаем нам статику напрямую
https://localhost/index.html
https://localhost/style.css
https://localhost/test.js

// запрос для проверки, что в случае неверного(несуществующего) файла нам будет отдан index.html
https://localhost/wrong

// можно, если сильно хочется, проверить еще back(например, через postman)
https://localhost/api



// такс, но стоит сказать, что все работать будет, но будет в браузере предупреждение, потому что он не доверяет нашим сертификатам, потому что они не подписаны центром сертификации
Поэтому для тех, кому этого не достаточно и кто хочет дойти до конца, придется немного исправить

mkcert localhost 127.0.0.1 ::1  // для создание своего центра сертификации
// затем нужно переименовать их как было раньше
mv cert.pem cert2.pem
mv key.pem key2.pem
mv localhost+2-key.pem key.pem
mv localhost+2.pem cert.pem

// еще нужно импортировать сертификат нашего CA, который в WSL, в Доверенные центры сертификации

// пересоздадим контейнеры
docker compose down
docker compose up --build -d

// после этого предупреждение должно пропасть
